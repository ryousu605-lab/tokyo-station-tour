```js
function openMap(){
  mapFloor = scenes[current].floor;
  renderMapFloor();
  document.getElementById("mapOverlay").classList.add("open");
  requestAnimationFrame(centerMapOnCurrent);
}

function setMapFloor(f){
  mapFloor = f;
  renderMapFloor();
}

function renderMapFloor(){
  const inner = document.getElementById("mapInner");
  const m = MAPS[mapFloor];
  document.getElementById("bigMapImg").src = m.image;

  // フロアタブの活性表示
  document.querySelectorAll(".floorTab").forEach(b =>
    b.classList.toggle("active", b.dataset.floor === mapFloor));

  // ピン: 表示フロアの地点のみ
  inner.querySelectorAll(".pin").forEach(p => p.remove());
  Object.entries(scenes).forEach(([id, s]) => {
    if(s.floor !== mapFloor) return;
    const pin = document.createElement("div");
    pin.className = "pin" + (s.dest ? " dest" : "") + (id === current ? " here" : "");
    pin.style.left = s.map.x + "%";
    pin.style.top = s.map.y + "%";
    pin.innerHTML = `<div class="dot"></div><div class="plabel">${s.name}</div>`;
    pin.onclick = () => { goTo(id); closeMap(); };
    inner.appendChild(pin);
  });

  // 接続線: 両端が表示フロアにあるものだけ描画（フロアをまたぐリンクは描かない）
  const svg = document.getElementById("trailSvg");
  let html = "";
  const seen = new Set();
  Object.entries(scenes).forEach(([id, s]) => {
    if(s.floor !== mapFloor) return;
    s.links.forEach(l => {
      if(scenes[l.to].floor !== mapFloor) return;
      const key = [id, l.to].sort().join("|");
      if(seen.has(key)) return;
      seen.add(key);
      const a = s.map, b = scenes[l.to].map;
      html += `<line x1="${a.x}" y1="${a.y}" x2="${b.x}" y2="${b.y}"
        stroke="rgba(92,102,80,0.55)" stroke-width="0.35" vector-effect="non-scaling-stroke" stroke-dasharray="1,1"/>`;
    });
  });

  // 軌跡: visited の隣接ペアのうち両端が表示フロアのものだけを、連続区間ごとの polyline に分割して描画
  let seg = [];
  const flush = () => {
    if(seg.length > 1){
      const pts = seg.map(id => `${scenes[id].map.x},${scenes[id].map.y}`).join(" ");
      html += `<polyline points="${pts}" fill="none" stroke="#c0392b"
        stroke-width="0.5" vector-effect="non-scaling-stroke" stroke-linejoin="round" stroke-linecap="round" opacity="0.85"/>`;
    }
    seg = [];
  };
  visited.forEach(id => {
    if(scenes[id].floor === mapFloor) seg.push(id);
    else flush();
  });
  flush();
  svg.innerHTML = html;

  document.getElementById("mHereLabel").textContent =
    scenes[current].floor === mapFloor ? "現在地: " + scenes[current].name : `${mapFloor} を表示中（現在地は${scenes[current].floor}）`;
}
```

### 4-3. 機能仕様（4機能。これ以外は実装しない）

共通: 取得座標はすべて小数1桁に丸める（`(v).toFixed(1)` → Number化）。取得のたびに `#editReadout` に表示し、`navigator.clipboard.writeText()` でコピーする（clipboard失敗時は無視して表示のみ）。

1. **写真上の座標取得** — `EDIT` 時のみ `#stage` に `click` リスナーを追加。クリック対象が `.hotspot` / `#compass` / `#minimap` / `#topbar` / `#howtoBtn` / `#howtoPanel` / `#editBar` の内部なら何もしない（`e.target.closest()` で判定）。それ以外なら `x = e.clientX/innerWidth*100`, `y = e.clientY/innerHeight*100` を取得し、クリップボードへ `x: 52.3, y: 61.0` 形式で書き込む。
2. **地図上の座標取得** — `EDIT` 時のみ `#mapInner` に `click` リスナーを追加。`.pin` 内部のクリックは無視。`getBoundingClientRect()` から%を計算し、`{x: 39.4, y: 79.7}` 形式でコピー。readout には `map(B1): {x: 39.4, y: 79.7}` のようにフロア名を添える。
3. **ホットスポットのドラッグ移動** — `EDIT` 時、各 `.hotspot` に Pointer Events（`pointerdown`/`pointermove`/`pointerup`, `setPointerCapture` 使用）でドラッグを実装。ドラッグ中は要素の `left/top` を追従させ、`pointerup` で `scenes[current].links[i].x / .y` に丸め値を書き戻し、readout に反映する。ドラッグ後の `click` によるシーン遷移を抑止するため、移動距離が 6px を超えたドラッグの直後は `onclick` を1回無効化する。
4. **`exportScenes()`** — 次の文字列を生成し、Blob（`text/javascript`）として `scenes.js` の名前でダウンロードさせる:
   ```js
   const text =
     "const MAPS = " + JSON.stringify(MAPS, null, 2) + ";\n" +
     "const startId = " + JSON.stringify(startId) + ";\n" +
     "const scenes = " + JSON.stringify(scenes, null, 2) + ";\n";
   ```
   ダウンロードは `<a download>` 方式（`URL.createObjectURL` → `a.click()` → `revokeObjectURL`）。

ピンのドラッグ（地図上の `map` 座標変更）は**実装しない**（機能2のクリック取得で代替。スコープ固定のため）。
